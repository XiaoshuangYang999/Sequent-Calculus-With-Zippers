\begin{document}

Note: ``Tree'' is GenT and ``Zip'' is GenZ, and ``L'' and ``R'' indicate the \texttt{foldl} and \texttt{foldr} variants.

\pgfplotsset{
  every axis/.style={
    font=\scriptsize,
    legend pos=north west, grid=major, width=6.3cm,
    ylabel=seconds, ymode=log, ymax=0.01,
    xlabel=formula length,
  }
}
\pgfplotsset{
  every axis legend/.append style={
    font=\small, legend image post style={scale=0.4},
    legend columns=-1,
    font=\scriptsize,
  }
}

\centering

% #1 = formula
% #2 = logic
\newcommand{\fourplot}[2]{
\begin{tikzpicture}
  \pgfplotstableread{results.csv.dat}\datatable
  \begin{axis}[xlabel=\textbf{#1 #2} formula length]
    \addlegendentry{TreeR} \addplot table[y=#1R-#2-GenT] from \datatable;
    \addlegendentry{ZipR}  \addplot table[y=#1R-#2-GenZ] from \datatable;
    \addlegendentry{TreeL} \addplot table[y=#1L-#2-GenT] from \datatable;
    \addlegendentry{ZipL}  \addplot table[y=#1L-#2-GenZ] from \datatable;
  \end{axis}
\end{tikzpicture}
}

% #1 = formula
% #2 = logic
\newcommand{\twoplot}[2]{
\begin{tikzpicture}
  \pgfplotstableread{results.csv.dat}\datatable
  \begin{axis}[xlabel=\textbf{#1 #2} formula length, ymax=0.001, xshift=6.2cm]
    \addlegendentry{Tree} \addplot table[y=#1-#2-GenT] from \datatable;
    \addlegendentry{Zip}  \addplot table[y=#1-#2-GenZ] from \datatable;
  \end{axis}
\end{tikzpicture}
}
